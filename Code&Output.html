def load_arff(file_path):
    # Load ARFF data
    data, b = arff.loadarff(file_path)
    # Convert to DataFrame
    df = pd.DataFrame(data)
    # Decode byte strings to UTF-8 
    for column in df.select_dtypes([object]).columns:
        df[column] = df[column].str.decode('utf-8')
    return df

# Add datasets to list so can iterate ovcer them
datasets = [load_arff(file_path1), load_arff(file_path2), load_arff(file_path3)]

# Function to plot the classifiers results
def plot_results(results, param_values, title, param_name):
    fig, axs = plt.subplots(1, 3, figsize=(18, 6))
    dataset_names = ['Ionosphere', 'Steel Plates Fault', 'Banknote Authentication']
    
    for i, (result, dataset_names) in enumerate(zip(results, dataset_names)):
        # Each result item is a list of accuracies for a specific parameter value
        data_for_plot = [np.array(acc) for acc in result]  # List of arrays for each parameter value
        
        # Make the boxplot
        axs[i].boxplot(data_for_plot, labels=param_values)
        axs[i].set_title(dataset_names)
        axs[i].set_xlabel(param_name)
        axs[i].set_ylabel('Accuracy')
        
    plt.suptitle(title)
    plt.show()

max_iterations = 50

# Function to run the classifier
def run_classifier(clasif_name, clasif_class, param_name, param_values, datasets):
    warnings.simplefilter("ignore") # Ignore silly warnings
    results = [] # List to store results
    # Loop over all 3 datasets
    for dataset in datasets:
        # Separate features and target  
        X = dataset.iloc[:, :-1].values
        y = dataset.iloc[:, -1].values

        dataset_results = [] # List to store results for each dataset
        # Loop over all parameter values for the classifier
        for param_value in param_values:
            accuracies = [] # List to store accuracies for each iteration
            # Loop over all iterations to calculate the average accuracy for the parameter value, should loop no more than 50 times
            for seed in range(max_iterations):
                try:
                    # Split data into training and test sets
                    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.5, random_state=seed)
                    
                    # Train and predict
                    classifier = clasif_class(**{param_name: param_value})
                    classifier.fit(X_train, y_train)

                    # Predict and calculate accuracy
                    y_pred = classifier.predict(X_test)
                    accuracies.append(accuracy_score(y_test, y_pred))
                
                # Catch error and give helpful error message
                except Exception as e:
                    print(f"Error with {clasif_name} and parameter {param_value} on iteration {seed}: {e}")
            dataset_results.append(accuracies)  # Append accuracies for the parameter value
        results.append(dataset_results)         # Append results for the dataset
    return results                              # Return results

# Initialise the DataFames for summary  tables
best_errors_table = pd.DataFrame(index=classifiers.keys(), columns=['Ionosphere', 'Steel Plates Fault', 'Banknote Auth'])
best_params_table = pd.DataFrame(index=classifiers.keys(), columns=['Ionosphere', 'Steel Plates Fault', 'Banknote Auth'])

# loop through the classifiers and results to find the best mean  eror and corresponding hyperparameter value
for clasif_name, results in zip(classifiers.keys(), all_results):
    # Get the parameter values for the current classifier
    param_values = list(params[clasif_name].values())[0]  
    # Loop through the datasets
    for i, dataset_results in enumerate(results):
        # Calculate the mean error for each parameter value by converting accuracy to error and find mean
        mean_errors = [np.mean(1 - np.array(accuracy)) for accuracy in dataset_results] 
        
        # Find the best mean error and corresponding hyperparameter value
        best_mean_error = min(mean_errors)
        
        # Get the best parameter value
        best_param_value = param_values[mean_errors.index(best_mean_error)]
        
        # Update tables
        dataset_name = ['Ionosphere', 'Steel Plates Fault', 'Banknote Auth'][i]
        best_errors_table.loc[clasif_name, dataset_name] = best_mean_error
        best_params_table.loc[clasif_name, dataset_name] = best_param_value

# Show summary tables
print("Summary Table 1: Lowest Mean Test Errors")
print(best_errors_table)

print("\nSummary Table 2: Corresponding Hyperparameter Values")
print(best_params_table)
